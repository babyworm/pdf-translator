# Phase 3: OCR Pipeline — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add OCR support so scanned PDFs (image-only, no text layer) can be translated. Auto-detect scanned vs text PDFs. Surya as primary OCR engine, Tesseract as lightweight fallback.

**Architecture:** OCR engines implement an `OCREngine` Protocol. The extractor auto-detects scanned pages and delegates to OCR. PDF builder uses background-color sampling (not redaction) for scanned pages.

**Tech Stack:** Python 3.10+, surya-ocr, pytesseract, Pillow

**Spec:** `docs/superpowers/specs/2026-04-01-pdf-translator-v2-design.md` §6, §3.3

**Prerequisites:** Phase 1 complete (core/ structure, pdf_builder with `is_scanned` parameter).

# pdf_translator/core/ocr/detector.py
from __future__ import annotations
import fitz


def is_scanned_page(page: fitz.Page, threshold: int = 50) -> bool:
    """Check if a page is scanned (image-only) by counting text characters."""
    text = page.get_text("text").strip()
    return len(text) < threshold


def detect_pdf_type(pdf_path: str) -> str:
    """Detect if PDF is 'text' or 'scanned' by sampling pages."""
    doc = fitz.open(pdf_path)
    try:
        if len(doc) == 0:
            return "scanned"
        scanned_count = 0
        sample_size = min(len(doc), 5)  # sample up to 5 pages
        for i in range(sample_size):
            if is_scanned_page(doc[i]):
                scanned_count += 1
        return "scanned" if scanned_count > sample_size / 2 else "text"
    finally:
        doc.close()
```

# pdf_translator/core/ocr/surya_engine.py
from __future__ import annotations
import logging
from pdf_translator.core.ocr.base import OCRResult

logger = logging.getLogger(__name__)


class SuryaOCREngine:
    name = "surya"

    def is_available(self) -> bool:
        try:
            import surya  # noqa: F401
            return True
        except ImportError:
            return False

    def extract(self, page_image: bytes, lang: str = "en") -> list[OCRResult]:
        try:
            from surya.ocr import run_ocr
            from surya.model.detection.model import load_model as load_det_model
            from surya.model.recognition.model import load_model as load_rec_model
            from PIL import Image
            import io

            image = Image.open(io.BytesIO(page_image))
            det_model = load_det_model()
            rec_model = load_rec_model()

            ocr_results = run_ocr(
                [image], [det_model, rec_model], langs=[[lang]],
            )

            results = []
            if ocr_results and len(ocr_results) > 0:
                for line in ocr_results[0].text_lines:
                    bbox = line.bbox  # [x0, y0, x1, y1]
                    results.append(OCRResult(
                        text=line.text,
                        bbox=list(bbox),
                        confidence=line.confidence,
                    ))
            return results
        except Exception as e:
            logger.warning("Surya OCR failed: %s", e)
            return []
```

# pdf_translator/core/ocr/tesseract_engine.py
from __future__ import annotations
import logging
import shutil
from pdf_translator.core.ocr.base import OCRResult

logger = logging.getLogger(__name__)

_LANG_MAP = {
    "en": "eng", "ko": "kor", "ja": "jpn", "zh": "chi_sim",
    "de": "deu", "fr": "fra", "es": "spa", "pt": "por", "it": "ita",
}


class TesseractOCREngine:
    name = "tesseract"

    def is_available(self) -> bool:
        return shutil.which("tesseract") is not None

    def extract(self, page_image: bytes, lang: str = "en") -> list[OCRResult]:
        try:
            import pytesseract
            from PIL import Image
            import io

            image = Image.open(io.BytesIO(page_image))
            tess_lang = _LANG_MAP.get(lang, "eng")

            data = pytesseract.image_to_data(
                image, lang=tess_lang, output_type=pytesseract.Output.DICT,
            )

            results = []
            n = len(data["text"])
            for i in range(n):
                text = data["text"][i].strip()
                conf = int(data["conf"][i])
                if not text or conf < 30:
                    continue
                x, y, w, h = data["left"][i], data["top"][i], data["width"][i], data["height"][i]
                results.append(OCRResult(
                    text=text,
                    bbox=[x, y, x + w, y + h],
                    confidence=conf / 100.0,
                ))
            return results
        except Exception as e:
            logger.warning("Tesseract OCR failed: %s", e)
            return []
```

from pdf_translator.core.ocr.surya_engine import SuryaOCREngine
from pdf_translator.core.ocr.tesseract_engine import TesseractOCREngine


def get_ocr_engine(name: str = "auto") -> OCREngine | None:
    """Get OCR engine by name or auto-detect."""
    if name == "surya":
        engine = SuryaOCREngine()
        return engine if engine.is_available() else None
    if name == "tesseract":
        engine = TesseractOCREngine()
        return engine if engine.is_available() else None
    if name == "auto":
        for EngineClass in [SuryaOCREngine, TesseractOCREngine]:
            engine = EngineClass()
            if engine.is_available():
                return engine
    return None
```

Add an `ocr_engine` parameter to `extract_pdf()`. When text extraction yields too few characters, fall back to OCR:

```python
def extract_pdf(pdf_path, output_dir=None, pages=None, ocr_engine=None):
    # ... existing extraction ...
    elements = parse_elements(data)

    # If too few elements and OCR engine available, try OCR
    if len(elements) < 3 and ocr_engine is not None:
        elements = _ocr_fallback(pdf_path, ocr_engine, pages)

    return elements
```

The `_ocr_fallback` function:
1. Opens PDF with fitz
2. For each page, renders to image (`page.get_pixmap().tobytes("png")`)
3. Calls `ocr_engine.extract(image_bytes, lang)`
4. Converts `OCRResult` list to `Element` list
